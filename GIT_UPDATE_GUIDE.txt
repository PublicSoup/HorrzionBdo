If you haven't set up git yet:

# 1. Configure git
git config --global user.name "Your Name"
git config --global user.email "your.email@example.com"

# 2. Initialize repository (if not already)
git init

# 3. Add remote repository (GitHub)
git remote add origin https://github.com/yourusername/HorrzionBdo.git

# Or if using SSH:
git remote add origin git@github.com:yourusername/HorrzionBdo.git

# 4. Check remote
git remote -v

# 5. Pull existing changes (if any)
git pull origin main

# 6. Then add your files (see above)

git status

# See diff of changes
git diff --staged

# See list of files to be committed
git diff --staged --name-only

# Remove a file from staging if needed
git reset HEAD filename.cpp
