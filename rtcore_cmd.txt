:: Copy these commands one by one into an Administrator Command Prompt

:: 1. Enable test signing
bcdedit /set testsigning on

:: 2. Disable driver blocklist
reg add "HKLM\SYSTEM\CurrentControlSet\Control\CI\Config" /v "VulnerableDriverBlocklistEnable" /t REG_DWORD /d 0 /f

:: 3. Copy RTCore64.sys to system directory (if you have it)
copy "C:\Program Files (x86)\MSI Afterburner\RTCore64.sys" "%SystemRoot%\System32\drivers\RTCore64.sys"

:: 4. Create and start the service
sc stop RTCore64
sc delete RTCore64
sc create RTCore64 type= kernel binPath= "%SystemRoot%\System32\drivers\RTCore64.sys"
sc start RTCore64

:: 5. Check service status
sc query RTCore64
